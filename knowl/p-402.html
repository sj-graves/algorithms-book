\begin{equation*} \CV{n} = \set{ \left.\cvec{z_0 \\ z_1 \\ \vdots \\ z_{n-1}} \right| z_0,z_1,\dotsc,z_{n-1}\in\Comps}\text{.} \end{equation*}
